Friendfollow SNS.widget.friendfollow

加关注组件

需额外引入：
http://a.tbcdn.cn/p/snsdk/core.js
snsdk核心文件

组件实现有三种方式：1、根据class钩子调用js; 2、js代码SNS.require()初始化调用; 3、js代码SNS.ui()外部统一调用使用;

源文件：friendfollow.js

作者：道璘

视觉规范


DEMO查看

Example Demo

三种方式引用

1、页面直接通过class="sns-widget"钩子引用组件,参数需要自定义属性data-friendfollow中注明

<span class="sns-widget" data-friendfollow='{"userId":2055117420, skinType:"green2"}'></span >

2、js代码SNS.require()异步调用

<span class="J_FriendFollow" data-userid="176010454" data-group-skin="green" ></span >

/**
 * 最里层API调用,针对复杂交换应用
 * @param {object} FriendFollow 组件对象 
 */
SNS.require(['sns/widget/follow/follow'], function(FriendFollow) {
	//更新所有页面内class为.J_FriendFollow组件
	FriendFollow.upload();

	FriendFollow.config('#J_Follow2', {
		srcNode : '#J_FriendGroupContainer',
		align : ['tr', 'br']
	});

	//容器统一配置, 针对容器内所有加关注按钮生效
	FriendFollow.configContainer('#container2', {
		align : ['tr', 'br'],
		//skin : 'green',
		follow : function(ev) {
			console.log('container2 follow ', ev)
		}
	});

	//容器统一绑定事件
	FriendFollow.on('#J_ToolbarContainer', 'follow', function(ev) {
		console.log(ev);
		//D.html('#info', '这里是关注回调事件, 次数为: ' + (++count));
	});
});
